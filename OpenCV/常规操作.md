### 1、读写视频

```python
import cv2

video_path = ''
video_save_path = ''

cap = cv2.VideoCapture(video_path)
size = (int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT)), int(cap.get(cv2.CAP_PROP_FRAME_WIDTH)))
fourcc = cv2.VideoWriter_fourcc(*'mp4v')
fps = cap.get(5)

writer = cv2.VideoWriter(video_save_path, fourcc, fps, size)

while True:
    ret, frame = cap.read()
    if ret:
    	writer.write(frame)
	else:
    	break
cap.release()
writer.release()
```
